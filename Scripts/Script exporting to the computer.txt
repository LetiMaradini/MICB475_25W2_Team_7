# Exporting qiime files from server to local computer

qiime tools export \
  --input-path /data/microgravity_project/table-no-mitochondria-no-chloroplast.qza \
  --output-path table-no-mitochondria-no-chloroplast_export

  --input-path /data/microgravity_project/taxonomy.qza \
